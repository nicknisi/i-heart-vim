Splits!

effortless spliting!


  function! WinMove(key)¬
      let t:curwin = winnr()¬
      exec "wincmd ".a:key¬
      if (t:curwin == winnr())¬
          if (match(a:key,'[jk]'))¬
              wincmd v¬
          else¬
              wincmd s¬
          endif¬
          exec "wincmd ".a:key¬
      endif¬
  endfunction¬

  map <silent> <C-h> :call WinMove('h')<cr>¬
  map <silent> <C-j> :call WinMove('j')<cr>¬
  map <silent> <C-k> :call WinMove('k')<cr>¬
  map <silent> <C-l> :call WinMove('l')<cr>¬
